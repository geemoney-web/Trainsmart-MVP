    Create `apps/api/src/tga/tga-scheduler.service.ts`:

    ```typescript
    import { Injectable, Logger } from '@nestjs/common';
    import { Cron } from '@nestjs/schedule';
    import { TgaSyncService } from './tga-sync.service';

    @Injectable()
    export class TgaSchedulerService {
      private readonly logger = new Logger(TgaSchedulerService.name);

      constructor(private readonly tgaSyncService: TgaSyncService) {}

      @Cron('0 16 * * *') // 4:00 PM UTC = 2:00 AM AEST
      async runNightlySync(): Promise<void> {
        this.logger.log('Nightly TGA sync starting...');
        try {
          const result = await this.tgaSyncService.syncAll('nightly_cron');
          this.logger.log(`Nightly sync complete: ${JSON.stringify(result)}`);
        } catch (error) {
          // Do NOT re-throw — scheduled jobs must not crash the process
          this.logger.error(`Nightly sync failed: ${error.message}`, error.stack);
        }
      }

      async triggerManualSync(userId: string): Promise<{ syncLogId: string }> {
        // Check for already-running sync
        const running = await this.tgaSyncService.getRunningSync();
        if (running) {
          throw new Error('A sync is already running');
        }
        // Start sync in background (don't await — return immediately)
        const syncLogId = await this.tgaSyncService.startSync(`manual:${userId}`);
        this.tgaSyncService.syncAll(`manual:${userId}`, syncLogId).catch(err => {
          this.logger.error(`Manual sync failed: ${err.message}`, err.stack);
        });
        return { syncLogId };
      }
    }
    ```

    Create `apps/api/src/tga/tga.controller.ts`:

    ```typescript
    import { Controller, Post, Get, Param, Req, ConflictException, HttpCode, HttpStatus } from '@nestjs/common';
    import { TgaSchedulerService } from './tga-scheduler.service';
    import { TgaSyncService } from './tga-sync.service';
    import { TgaApiClient } from './tga-api.client';
    import { Request } from 'express';

    @Controller('tga')
    export class TgaController {
      constructor(
        private readonly scheduler: TgaSchedulerService,
        private readonly syncService: TgaSyncService,
        private readonly tgaApiClient: TgaApiClient,
      ) {}

      // Manual sync trigger
      @Post('sync/trigger')
      @HttpCode(HttpStatus.ACCEPTED)
      async triggerSync(@Req() req: Request) {
        const userId = (req.user as any)?.sub ?? 'unknown';
        try {
          return await this.scheduler.triggerManualSync(userId);
        } catch (error) {
          if (error.message === 'A sync is already running') {
            throw new ConflictException('A sync is already in progress');
          }
          throw error;
        }
      }

      // Sync status polling
      @Get('sync/status/:syncLogId')
      async getSyncStatus(@Param('syncLogId') syncLogId: string) {
        return this.syncService.getSyncStatus(syncLogId);
      }

      // Sync history
      @Get('sync/history')
      async getSyncHistory() {
        return this.syncService.getSyncHistory(10);
      }

      // TGA qualification search (proxied to avoid CORS issues from frontend)
      @Get('qualifications/search')
      async searchQualifications(@Req() req: Request) {
        const query = (req.query.q as string) ?? '';
        if (!query || query.length < 2) return [];
        return this.tgaApiClient.searchQualifications(query);
      }
    }
    ```
